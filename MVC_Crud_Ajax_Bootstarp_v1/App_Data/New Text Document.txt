Create Procedure [dbo].[InsertUpdateEmployee](
	@ID Integer,
	@Name Nvarchar(100),
	@Age Integer,
	@State Nvarchar(100),
	@Country Nvarchar(100),
	@Action Varchar(50)
)
AS
	Begin
		IF @Action='Insert'
			Begin
				INSERT INTO Employee
				([Name],[Age],[Country],[State])
				Values (@Name, @Age, @Country, @State)
			End
		IF @Action = 'Update'
			Begin
				Update Employee
					SET [Name] = @Name,
					Age = @Age,
					Country = @Country,
					[State] = @State
					Where ID = @ID
				End
	End
GO
